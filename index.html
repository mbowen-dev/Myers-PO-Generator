  <script>

    function generatePO() {

      const eventInput = document.getElementById('eventNumber').value.trim().toUpperCase();

      const vendorCode = document.getElementById('vendor').value;

      const quoteNumber = document.getElementById('quoteNumber').value;

      const error = document.getElementById('error');

      const result = document.getElementById('result');

      const poNumber = document.getElementById('poNumber');

      const success = document.getElementById('success');


      error.style.display = 'none';

      success.style.display = 'none';

      result.style.display = 'none';


      if (!eventInput) {

        showError('Please enter an event number.');

        return;

      }


      if (!/^E-\d+$/.test(eventInput)) {

        showError('Event number must start with E- followed by numbers only. Example: E-1025');

        return;

      }


      if (!vendorCode) {

        showError('Please select a vendor.');

        return;

      }


      if (!quoteNumber || Number(quoteNumber) < 1 || !Number.isInteger(Number(quoteNumber))) {

        showError('Please enter a whole number of 1 or higher for the vendor quote number.');

        return;

      }


      const formattedQuoteNumber = String(quoteNumber);

      const generatedPO = `${eventInput}-${vendorCode}-${formattedQuoteNumber}`;


      poNumber.textContent = generatedPO;

      result.style.display = 'block';

    }


    function showError(message) {

      const error = document.getElementById('error');

      error.textContent = message;

      error.style.display = 'block';

    }


    async function copyPO() {

      const poNumber = document.getElementById('poNumber').textContent;

      const success = document.getElementById('success');


      success.style.display = 'none';


      if (!poNumber) {

        alert('Please generate a PO number first.');

        return;

      }


      try {

        if (navigator.clipboard && window.isSecureContext) {

          await navigator.clipboard.writeText(poNumber);

        } else {

          const textArea = document.createElement('textarea');

          textArea.value = poNumber;

          textArea.style.position = 'fixed';

          textArea.style.left = '-999999px';

          textArea.style.top = '-999999px';

          document.body.appendChild(textArea);

          textArea.focus();

          textArea.select();

          document.execCommand('copy');

          document.body.removeChild(textArea);

        }


        success.textContent = 'Copied to clipboard!';

        success.style.display = 'block';

      } catch (err) {

        const textArea = document.createElement('textarea');

        textArea.value = poNumber;

        document.body.appendChild(textArea);

        textArea.focus();

        textArea.select();

        alert('Copy did not work automatically. The PO number is selected now, so you can copy it manually.');

      }

    }
